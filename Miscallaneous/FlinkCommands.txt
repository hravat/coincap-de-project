CREATE TABLE kafka_source (
    id STRING,
    symbol STRING,
    rateUsd STRING,  
    source STRING,
    event_timestamp STRING
) WITH (
    'connector' = 'kafka',
    'topic' = 'test-topic',
    'properties.bootstrap.servers' = 'broker:29092',
    'properties.group.id' = 'flink-consumer',
    'scan.startup.mode' = 'latest-offset',
    'format' = 'json',
    'json.ignore-parse-errors' = 'true'
);
